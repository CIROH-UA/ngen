638 inline void init_deferred_associations() {
639 for (int d = 0; d < deferredProviders.size(); ++d) {
640 std::shared_ptr<data_access::OptionalWrappedDataProvider> &deferredProvider = deferredProviders[d];
641 // Skip doing anything for any deferred provider that already has its backing provider set
642 if (deferredProvider->isWrappedProviderSet())
643 continue;
644
645 // TODO: improve this later; since defaults can be used, it is technically possible to grab something
646 // valid when something more appropriate would later be available
647 // Iterate through available data providers and set association once a sufficient one is found
648 std::map<std::string, std::shared_ptr<data_access::GenericDataProvider>>::iterator avail_it;
649 for (avail_it = availableData.begin(); avail_it != availableData.end(); avail_it++) {
650 // If this satisfies everything this deferred provider needs to provide, and thus can be/was set ...
651 if (deferredProvider->setWrappedProvider(avail_it->second.get())) {
652 break;
653 }
654 }
655
656 // If none of the available data providers could be used for this deferred one, throw exception
657 if (!deferredProvider->isWrappedProviderSet()) {
658 // TODO: this probably needs to be some kind of custom configuration exception
659 std::string msg = "Multi BMI formulation cannot be created from config: cannot find available data "
660 "provider to satisfy set of deferred provisions for nested module at index "
661 + std::to_string(deferredProviderModuleIndices[d]) + ": {";
662 // There must always be at least 1; get manually to help with formatting
663 msg += deferredProvider->get_available_variable_names()[0];
664 // And here make sure to start at 1 instead of 0
665 for (int i = 1; i < deferredProvider->get_available_variable_names().size(); ++i)
666 msg += ", " + deferredProvider->get_available_variable_names()[i];
667 msg += "}";
668 throw realization::ConfigurationException(msg);
669 }
670 }

688 template<class T>
689 std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
690 std::shared_ptr<data_access::GenericDataProvider> wfp = std::make_shared<data_access::WrappedDataProvider>(this);
691 std::shared_ptr<T> mod = std::make_shared<T>(identifier, wfp, output);
692
693 // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
694 Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
695 "{{id}}", id);
696
697 // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
698 mod->create_formulation(properties);
699
700 // Set this up for placing in the module_variable_maps member variable
701 std::shared_ptr<std::map<std::string, std::string>> var_aliases;
702 var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
703 for (const std::string &var_name : mod->get_bmi_input_variables()) {
704 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
705 (*var_aliases)[framework_alias] = var_name;
706 // If framework_name is not yet in collection from which we have available data sources ...
707 if (availableData.count(framework_alias) != 1) {
708 setup_nested_deferred_provider(var_name, framework_alias, mod, mod_index);
709 }
710 else {
711 mod->input_forcing_providers[var_name] = availableData[framework_alias];
712 mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
713 }
714 }
715
716 // Also add the output variable aliases
717 for (const std::string &var_name : mod->get_bmi_output_variables()) {
718 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
719 (*var_aliases)[framework_alias] = var_name;
720 if (availableData.count(framework_alias) > 0) {
721 throw std::runtime_error(
722 "Multi BMI cannot be created with module " + mod->get_model_type_name() +
723 " with output variable " + framework_alias +
724 (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
725 " because a previous module is using this output variable name/alias.");
726 }
727 availableData[framework_alias] = mod;
728 }
729 module_variable_maps[mod_index] = var_aliases;
730 return mod;
731 }

763 std::shared_ptr<T> mod,

765 // TODO: probably don't actually need bmi_input_var_name, and just can deal with framework_output_name

767 // Only include BMI variable name, as that's what'll be visible when associating to backing provider
768 // It's "deferred" in that we'll set the backing later.
769 // It's "optional" in that it waits to use backing provider, using the default some number of times
770 std::shared_ptr<data_access::OptionalWrappedDataProvider> provider;
771 // TODO: make sure only alias is needed
772 auto defs_it = default_output_values.find(framework_output_name);
773 if (defs_it != default_output_values.end()) {
774 // TODO: consider also reading wait count from config

778 provider = std::make_shared<data_access::OptionalWrappedDataProvider>(framework_output_name);
779 }
780
781 // Add deferred to collection and module index to collection
782 deferredProviders.push_back(provider);
783 deferredProviderModuleIndices.push_back(mod_index);
784 // Assign as provider within module
785 // TODO: per TODO at top, probably can replace bmi_input_var_name here with framework_output_name
786 mod->input_forcing_providers[bmi_input_var_name] = provider;
787 }
